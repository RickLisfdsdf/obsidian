<%*
// 1. 收集人工输入的信息 (开启多行输入支持)
// tp.system.prompt 参数：(提示语, 默认值, 是否抛出取消异常, 是否允许多行)
const rawText = await tp.system.prompt("请粘贴完整的抖音分享文案:", "", false, true);

if (!rawText) {
    new Notice("未输入文案，操作取消。");
    return;
}

const summary = await tp.system.prompt("请输入你的总结 (支持多行/Markdown格式):", "", false, true);

if (!summary) {
    new Notice("未输入总结内容，操作取消。");
    return;
}

// ================= 本地清洗逻辑 =================
// 提取纯净 URL
const urlMatch = rawText.match(/https?:\/\/v\.douyin\.com\/[a-zA-Z0-9_-]+\/?/);
const cleanUrl = urlMatch ? urlMatch[0] : rawText; 

// 提取文案自带的标签 (Tags)
const tags = [];
const tagRegex = /#\s*([^\s#]+)/g;
let match;
while ((match = tagRegex.exec(rawText)) !== null) {
    tags.push(match[1].trim());
}
// =====================================================

new Notice("正在调用 DeepSeek API 提取关联概念...");

// 2. 配置 API 请求 (DeepSeek 格式)
const API_KEY = "sk-45591e684b8c44d1a27824f61db6be2f"; // 你的 DeepSeek API Key
const MODEL_NAME = "deepseek-chat"; 
const API_URL = "https://api.deepseek.com/chat/completions";

let conceptsText = "- [[待提取概念]]"; // 默认占位符

try {
    const response = await fetch(API_URL, {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json',
            'Authorization': `Bearer ${API_KEY}`
        },
        body: JSON.stringify({
            model: MODEL_NAME,
            messages: [
                {
                    role: "system",
                    content: "你是一个专业的知识管理助手。请从用户输入的总结中，提取3-5个最核心的专业名词、方法论或工具名称。严格只输出这些名词，用英文逗号分隔，不要包含任何其他说明文字。"
                },
                {
                    role: "user",
                    content: summary
                }
            ],
            temperature: 0.1 
        })
    });

    const data = await response.json();
    
    // 3. 处理 DeepSeek 返回的结果
    if (data.choices && data.choices.length > 0) {
        const rawConcepts = data.choices[0].message.content;
        
        // 清理换行符、中文逗号等，转换为数组
        const conceptArray = rawConcepts
            .replace(/\n/g, "")
            .replace(/，/g, ",")
            .split(",")
            .map(c => c.trim())
            .filter(c => c !== "");

        // 拼接成 Obsidian 的列表和双链语法
        if (conceptArray.length > 0) {
             conceptsText = conceptArray.map(c => `- [[${c}]]`).join('\n');
             new Notice("概念提取成功！");
        } else {
             new Notice("API 未返回有效概念。");
        }
    } else {
        console.error("DeepSeek 错误响应:", data);
        let errMsg = "未知错误";
        if (data.error && data.error.message) {
            errMsg = data.error.message;
        }
        new Notice(`API 拒绝请求: ${errMsg}`, 10000);
    }
} catch(error) {
    console.error("网络请求错误:", error);
    new Notice("网络请求失败，已保留占位符，请查阅控制台排错。");
}

// 4. 获取当前日期
const today = tp.date.now("YYYY-MM-DD");

// 5. 拼接最终的 Markdown 模板
const template = `---
url: ${cleanUrl}
author: 抖音视频
date: ${today}
tags:
${tags.length > 0 ? tags.map(t => `  - ${t}`).join('\n') : '  - 未分类'}
---

## 我的总结记录
${summary}

## 关联概念
${conceptsText}
`;

// 6. 输出到当前笔记
tR += template;
%>
