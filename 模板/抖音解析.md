<%*
// 1. 弹出输入框，让你粘贴原始的乱码文案
const rawText = await tp.system.prompt("请粘贴抖音分享文案:");

if (rawText) {
    // 2. 正则提取 URL
    const urlMatch = rawText.match(/https?:\/\/v\.douyin\.com\/[a-zA-Z0-9]+\//);
    const url = urlMatch ? urlMatch[0] : "";

    // 3. 正则提取标签 (Tags)
    const tags = [];
    const tagRegex = /#\s*([^\s#]+)/g;
    let match;
    while ((match = tagRegex.exec(rawText)) !== null) {
        tags.push(match[1].trim());
    }

    // 4. 清理废话并提取核心标题
    // 剔除 URL、引流话术和标签
    let title = rawText
        .replace(url, "")
        .replace(/复制此链接.*/g, "")
        .replace(tagRegex, "")
        .trim();
    
    // 过滤开头的英文字母/数字乱码，从第一个汉字开始截取
    const zhMatch = title.match(/[\u4e00-\u9fa5].*/);
    if (zhMatch) {
        title = zhMatch[0].trim();
    }

    // 5. 获取当前日期
    const today = tp.date.now("YYYY-MM-DD");

    // 6. 拼接最终的 Markdown 结构
    const template = `---
url: ${url}
author: 抖音视频
date: ${today}
tags:
${tags.length > 0 ? tags.map(t => `  - ${t}`).join('\n') : '  - 未分类'}
---

# ${title || "在此输入核心议题"}

## 我的总结记录


## 关联概念
- 
`;
    // 输出结果
    tR += template;
} else {
    tR += "未输入任何内容。";
}
%>
